4   import io.vavr.control.Try;

6   /**
7    * This record represents The Codest employee on Slack.
8    *
9    * @param slackId
10   *            internal Slack id.
11   * @param name
12   *            full name as it is set on Slack.
13   * @param email
14   *            email as it is set on Slack.
15   */
16  public record SlackUser(String slackId, String name, String email) {
17  
18  	/**
19  	 * Return user email in normalized form. See @{@link EmailNormalization}.
20  	 *
21  	 * @return normalized email
22  	 */
23  	public String normalizedEmail() {
24  		return Try.success(email())
25  				.mapTry(EmailNormalization::normalize)
26  				.map(EmailNormalization.NormalizedEmail::normalForm)
27  				.getOrElseThrow(t -> new RuntimeException(t));
28  	}
29  }
