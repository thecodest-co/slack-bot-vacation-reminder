3   /**
4    * For requirement, run on second working day of first month of each quarter.
5    *
6    * <p>This is very tricky and impossible to implement without complex calendar
7    * support. So, let's make some assumptions.
8    *
9    * <ul>
10   * <li>We omit all midweek holidays » to implement that feature we need to know
11   * holiday calendar
12   * for every single country, region and ethnic group.</li>
13   * <li>We use Gregorian Calendar with saturday and sunday as days off » other
14   * week model need to implement
15   * different calendar support.</li>
16   * </ul>
17   *
18   * <p>Based on that we can define some rules.
19   *
20   * <ul>
21   * <li>Only valid „run months” are January, April, July and October.</li>
22   * <li>Only valid „run days” are between 2nd and 4th and are not Saturday or
23   * Sunday.</li>
24   * <li>2nd is valid only if it is between Tuesday and Friday » 1st is Monday to
25   * Thursday</li>
26   * <li>3rd is valid only if it is Tuesday » 2st is Monday</li>
27   * <li>4th is valid only if it is Monday or Tuesday</li>
28   * <li>All other days are „invalid”.</li>
29   * </ul>
30   *
31   * <p>Example dates that match cases above:
32   *
33   * <ul>
34   * <li>January 2019 – 2nd is Wednesday</li>
35   * <li>April 2019 – 2nd is Tuesday</li>
36   * <li>January 2020 – 2nd is Thursday</li>
37   * <li>April 2021 – 2nd is Friday</li>
38   * <li>April 2018 – 3rd is Tuesday</li>
39   * <li>October 2022 – 4th is Tuesday</li>
40   * <li>July 2022 – 4th is Monday</li>
41   * </ul>
42   */
43  public interface RunningCalculator {
44  	/**
45  	 * Check if current local date is valid.
46  	 *
47  	 * @return true if date match to the rules. Otherwise, return false.
48  	 */
49  	boolean shouldItRun();
50  }
