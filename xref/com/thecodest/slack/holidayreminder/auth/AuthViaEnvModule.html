1   package com.thecodest.slack.holidayreminder.auth;
2   
3   import com.google.inject.AbstractModule;
4   import com.google.inject.name.Names;
5   
6   /**
7    * Module configure authentication for Slack and Calamari.
8    *
9    * <p>Looking for:
10   * <ul>
11   * <li>SLACK_TOKEN</li>
12   * <li>CALAMARI_USERNAME</li>
13   * <li>CALAMARI_PASSWORD</li>
14   * </ul>>
15   * Environment variables.
16   */
17  public class AuthViaEnvModule extends AbstractModule {
18  
19  	/**
20  	 * Read environment variables and set them into named Strings.
21  	 * <ul>
22  	 * <li>SLACK_TOKEN » slack-token</li>
23  	 * <li>CALAMARI_USERNAME » calamari-username</li>
24  	 * <li>CALAMARI_PASSWORD » calamari-password</li>
25  	 * </ul>>
26  	 */
27  	@Override
28  	protected void configure() {
29  		bind(String.class)
30  				.annotatedWith(Names.named("slack-token"))
31  				.toInstance(System.getenv("SLACK_TOKEN"));
32  		bind(String.class)
33  				.annotatedWith(Names.named("calamari-username"))
34  				.toInstance(System.getenv("CALAMARI_USERNAME"));
35  		bind(String.class)
36  				.annotatedWith(Names.named("calamari-password"))
37  				.toInstance(System.getenv("CALAMARI_PASSWORD"));
38  	}
39  }
