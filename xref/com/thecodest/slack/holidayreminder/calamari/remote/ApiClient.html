836 	 * @param response
837 	 *            An instance of the Response object
838 	 * @return Prepared file for the download
839 	 * @throws IOException
840 	 *             If fail to prepare file for download

842 	public File prepareDownloadFile(Response response) throws IOException {
843 		String filename = null;
844 		String contentDisposition = response.header("Content-Disposition");
845 		if (contentDisposition != null && !"".equals(contentDisposition)) {
846 			// Get filename from the Content-Disposition header.
847 			Pattern pattern = Pattern.compile("filename=['\"]?([^'\"\\s]+)['\"]?");
848 			Matcher matcher = pattern.matcher(contentDisposition);
849 			if (matcher.find()) {
850 				filename = sanitizeFilename(matcher.group(1));
851 			}
852 		}
853 
854 		String prefix = null;
855 		String suffix = null;
856 		if (filename == null) {
857 			prefix = "download-";
858 			suffix = "";
859 		} else {
860 			int pos = filename.lastIndexOf(".");
861 			if (pos == -1) {
862 				prefix = filename + "-";
863 			} else {
864 				prefix = filename.substring(0, pos) + "-";
865 				suffix = filename.substring(pos);
866 			}
867 			// File.createTempFile requires the prefix to be at least three characters long
868 			if (prefix.length() < 3)
869 				prefix = "download-";
870 		}
871 
872 		if (tempFolderPath == null)
873 			return Files.createTempFile(prefix, suffix).toFile();
874 		else
875 			return Files.createTempFile(Paths.get(tempFolderPath), prefix, suffix).toFile();
876 	}

975 	 * @param response
976 	 *            Response
977 	 * @param returnType
978 	 *            Return type
979 	 * @return Type
980 	 * @throws ApiException
981 	 *             If the response has a unsuccessful status code or
982 	 *             fail to deserialize the response body
983 	 */
984 	public <T> T handleResponse(Response response, Type returnType) throws ApiException {
985 		if (response.isSuccessful()) {
986 			if (returnType == null || response.code() == 204) {
987 				// returning null if the returnType is not defined,
988 				// or the status code is 204 (No Content)
989 				if (response.body() != null) {
990 					try {
991 						response.body().close();
992 					} catch (IOException e) {
993 						throw new ApiException(response.message(), e, response.code(), response.headers().toMultimap());
994 					}
995 				}
996 				return null;
997 			} else {
998 				return deserialize(response, returnType);
999 			}
1000 		} else {
1001 			String respBody = null;
1002 			if (response.body() != null) {
1003 				try {
1004 					respBody = response.body().string();
1005 				} catch (IOException e) {
1006 					throw new ApiException(response.message(), e, response.code(), response.headers().toMultimap());
1007 				}
1008 			}
1009 			throw new ApiException(response.message(), response.code(), response.headers().toMultimap(), respBody);
1010 		}
1011 	}

1053 	 * @param method
1054 	 *            The request method, one of "GET", "HEAD", "OPTIONS", "POST",
1055 	 *            "PUT", "PATCH" and "DELETE"
1056 	 * @param queryParams
1057 	 *            The query parameters
1058 	 * @param collectionQueryParams
1059 	 *            The collection query parameters
1060 	 * @param body
1061 	 *            The request body object
1062 	 * @param headerParams
1063 	 *            The header parameters
1064 	 * @param formParams
1065 	 *            The form parameters
1066 	 * @param authNames
1067 	 *            The authentications to apply
1068 	 * @param progressRequestListener
1069 	 *            Progress request listener
1070 	 * @return The HTTP request
1071 	 * @throws ApiException
1072 	 *             If fail to serialize the request body object
1073 	 */
1074 	public Request buildRequest(String path, String method, List<Pair> queryParams, List<Pair> collectionQueryParams,
1075 			Object body, Map<String, String> headerParams, Map<String, Object> formParams, String[] authNames,
1076 			ProgressRequestBody.ProgressRequestListener progressRequestListener) throws ApiException {
1077 		updateParamsForAuth(authNames, queryParams, headerParams);
1078 
1079 		final String url = buildUrl(path, queryParams, collectionQueryParams);
1080 		final Request.Builder reqBuilder = new Request.Builder().url(url);
1081 		processHeaderParams(headerParams, reqBuilder);
1082 
1083 		String contentType = headerParams.get("Content-Type");
1084 		// ensuring a default content type
1085 		if (contentType == null) {
1086 			contentType = "application/json";
1087 		}
1088 
1089 		RequestBody reqBody;
1090 		if (!HttpMethod.permitsRequestBody(method)) {
1091 			reqBody = null;
1092 		} else if ("application/x-www-form-urlencoded".equals(contentType)) {
1093 			reqBody = buildRequestBodyFormEncoding(formParams);
1094 		} else if ("multipart/form-data".equals(contentType)) {
1095 			reqBody = buildRequestBodyMultipart(formParams);
1096 		} else if (body == null) {
1097 			if ("DELETE".equals(method)) {
1098 				// allow calling DELETE without sending a request body
1099 				reqBody = null;
1100 			} else {
1101 				// use an empty request body (for POST, PUT and PATCH)
1102 				reqBody = RequestBody.create(MediaType.parse(contentType), "");
1103 			}
1104 		} else {
1105 			reqBody = serialize(body, contentType);
1106 		}
1107 
1108 		Request request = null;
1109 
1110 		if (progressRequestListener != null && reqBody != null) {
1111 			ProgressRequestBody progressRequestBody = new ProgressRequestBody(reqBody, progressRequestListener);
1112 			request = reqBuilder.method(method, progressRequestBody).build();
1113 		} else {
1114 			request = reqBuilder.method(method, reqBody).build();
1115 		}
1116 
1117 		return request;
1118 	}

1120 	/**
1121 	 * Build full URL by concatenating base path, the given sub path and query
1122 	 * parameters.
1123 	 *
1124 	 * @param path
1125 	 *            The sub path
1126 	 * @param queryParams
1127 	 *            The query parameters
1128 	 * @param collectionQueryParams
1129 	 *            The collection query parameters
1130 	 * @return The full URL
1131 	 */
1132 	public String buildUrl(String path, List<Pair> queryParams, List<Pair> collectionQueryParams) {
1133 		final StringBuilder url = new StringBuilder();
1134 		url.append(basePath).append(path);
1135 
1136 		if (queryParams != null && !queryParams.isEmpty()) {
1137 			// support (constant) query string in `path`, e.g. "/posts?draft=1"
1138 			String prefix = path.contains("?") ? "&" : "?";
1139 			for (Pair param : queryParams) {
1140 				if (param.getValue() != null) {
1141 					if (prefix != null) {
1142 						url.append(prefix);
1143 						prefix = null;
1144 					} else {
1145 						url.append("&");
1146 					}
1147 					String value = parameterToString(param.getValue());
1148 					url.append(escapeString(param.getName())).append("=").append(escapeString(value));
1149 				}
1150 			}
1151 		}
1152 
1153 		if (collectionQueryParams != null && !collectionQueryParams.isEmpty()) {
1154 			String prefix = url.toString().contains("?") ? "&" : "?";
1155 			for (Pair param : collectionQueryParams) {
1156 				if (param.getValue() != null) {
1157 					if (prefix != null) {
1158 						url.append(prefix);
1159 						prefix = null;
1160 					} else {
1161 						url.append("&");
1162 					}
1163 					String value = parameterToString(param.getValue());
1164 					// collection query parameter value already escaped as part of parameterToPairs
1165 					url.append(escapeString(param.getName())).append("=").append(value);
1166 				}
1167 			}
1168 		}
1169 
1170 		return url.toString();
1171 	}

1270 	 * verifyingSsl and sslCaCert.

1272 	private void applySslSettings() {

1274 			TrustManager[] trustManagers = null;
1275 			HostnameVerifier hostnameVerifier = null;
1276 			if (!verifyingSsl) {
1277 				TrustManager trustAll = new X509TrustManager() {

1290 						return null;

1294 				trustManagers = new TrustManager[]{trustAll};

1298 						return true;
1299 					}
1300 				};

1302 				char[] password = null; // Any password will work.

1304 				Collection<? extends Certificate> certificates = certificateFactory.generateCertificates(sslCaCert);
1305 				if (certificates.isEmpty()) {

1312 					caKeyStore.setCertificateEntry(certificateAlias, certificate);

1315 						.getInstance(TrustManagerFactory.getDefaultAlgorithm());
1316 				trustManagerFactory.init(caKeyStore);
1317 				trustManagers = trustManagerFactory.getTrustManagers();

1319 
1320 			if (keyManagers != null || trustManagers != null) {
1321 				SSLContext sslContext = SSLContext.getInstance("TLS");
1322 				sslContext.init(keyManagers, trustManagers, new SecureRandom());
1323 				httpClient.setSslSocketFactory(sslContext.getSocketFactory());
