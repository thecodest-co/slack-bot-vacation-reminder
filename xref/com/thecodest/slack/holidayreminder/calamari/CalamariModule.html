1   package com.thecodest.slack.holidayreminder.calamari;
2   
3   import com.google.inject.AbstractModule;
4   import com.google.inject.Provides;
5   import com.thecodest.slack.holidayreminder.calamari.remote.ApiClient;
6   import com.thecodest.slack.holidayreminder.calamari.remote.api.AbsenceTypeApi;
7   import com.thecodest.slack.holidayreminder.calamari.remote.api.EmployeesApi;
8   import javax.inject.Named;
9   import javax.inject.Singleton;
10  
11  /**
12   * Setup of Calamari module. We looking for named Strings
13   * <samp>calamari-username</samp>
14   * and <samp>calamari-username</samp>.
15   */
16  public class CalamariModule extends AbstractModule {
17  
18  	/**
19  	 * Configuration of @{@link CalamariApi} interface.
20  	 */
21  	@Override
22  	protected void configure() {
23  		bind(CalamariApi.class).to(SimpleCalamariApi.class);
24  	}
25  
26  	/**
27  	 * Main {@link ApiClient} provider.
28  	 *
29  	 * @param username
30  	 *            named String <samp>calamari-username</samp>.
31  	 * @param password
32  	 *            named String <samp>calamari-password</samp>.
33  	 * @return instance that is singleton.
34  	 */
35  	@Provides
36  	@Singleton
37  	public ApiClient apiClient(@Named("calamari-username") String username,
38  			@Named("calamari-password") String password) {
39  		var apiClient = new ApiClient();
40  		apiClient.setUsername(username);
41  		apiClient.setPassword(password);
42  		return apiClient;
43  	}
44  
45  	/**
46  	 * Provides {@link EmployeesApi} client instance.
47  	 *
48  	 * @param apiClient
49  	 *            main calamari client.
50  	 * @return instance that is singleton.
51  	 */
52  	@Provides
53  	@Singleton
54  	public EmployeesApi employeesApi(ApiClient apiClient) {
55  		return new EmployeesApi(apiClient);
56  	}
57  
58  	/**
59  	 * Provides {@link AbsenceTypeApi} client instance.
60  	 *
61  	 * @param apiClient
62  	 *            main calamari client.
63  	 * @return instance that is singleton.
64  	 */
65  	@Provides
66  	@Singleton
67  	public AbsenceTypeApi absenceTypeApi(ApiClient apiClient) {
68  		return new AbsenceTypeApi(apiClient);
69  	}
70  
71  }
